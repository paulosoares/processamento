Registrar Andamento Indevido
store "600000179", "600000628", "600000179" setores
store Massa/processosLancamentoIndevido.xml massa
foreach setor ${setores}
loadVars ../setores.xml codigoSetor==${setor}
call alterarSetor
open /processamento/pages/andamento/registrarAndamento.jsf
forXml ${massa}
type id=inputClasse ${classe}
type id=inputProcesso ${processo}
click id=btPesquisar
waitForVisible //span[@id="pnlAndamentosProcesso"]
storeElementIndex //span[@id="pnlAndamentosProcesso"]//tbody/tr[last()] ultimoRegistro
storeEval ${ultimoRegistro}+1 ultimoRegistro
storeText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]/td[2] dataAndamento
storeText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]/td[3] codAndamento
storeText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]/td[4] descAndamento
waitForVisible //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]//img[@title="Lançar andamento indevido"]
storeEval window.scrollBy(0,2000) mostraOFimDaPagina
click //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]//img[@title="Lançar andamento indevido"]
waitForElementPresent //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}+1]
storeElementIndex //span[@id="pnlAndamentosProcesso"]//tbody/tr[last()] ultimoRegistro
storeEval ${ultimoRegistro}+1 sequencia
storeEval var d = new Date(); var dia = d.getDate(); var mes = d.getMonth() + 1; var ano = d.getFullYear(); dia + "/" + ((mes<10)?"0"+mes:mes) + "/" + ano; dataHoje
assertText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${sequencia}]/td[1] ${sequencia}
assertText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${sequencia}]/td[2] ${dataHoje}
assertText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${sequencia}]/td[3] 7700
assertText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${sequencia}]/td[8] ${dataAndamento} - ${descAndamento}
click //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[last()]//img[@title="Desfazer andamento indevido"]
waitForElementNotPresent //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${sequencia}]/td[1]
assertText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]/td[1] ${ultimoRegistro}
assertText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]/td[2] ${dataAndamento}
assertText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]/td[3] ${codAndamento}
assertText //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]/td[4] ${descAndamento}
assertElementPresent //span[@id="pnlAndamentosProcesso"]/table/tbody/tr[${ultimoRegistro}]//img[@title="Lançar andamento indevido"]
endForXml
endForeach
script alterarSetor
store ${codigoSetor} codigoSetor
store ${siglaSetor} siglaSetor
open /processamento/pages/principal/principal.jsf
storeText //html/body/form/table[2]/tbody/tr/td[2]/table/tbody/tr/td nomeUsuarioComSetorAtual
storeEval "${nomeUsuarioComSetorAtual}".split("@")[0] nomeUsuario
storeEval "${nomeUsuarioComSetorAtual}".split("@")[1] setorAtual
select seletorSetores label=*${codigoSetor}*
click //span[@id='pnlPrincipalPesquisa']/table/tbody/tr/td[3]/input
waitForTextPresent Setor alterado com sucesso!
assertTextPresent ${siglaSetor}
endScript
